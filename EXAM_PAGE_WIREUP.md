# Exam Page Integration Guide

This guide shows how to wire up all the implemented hooks and components into the exam-taking page (`src/app/student/exam/[id]/page.tsx`).

## Step 1: Import All Required Hooks and Components

```typescript
'use client';

import { useRef, useState, useEffect, useCallback } from 'react';
import { useRouter } from 'next/navigation';
import { useExamTimer } from '@/hooks/useExamTimer';
import { useAutoSave } from '@/hooks/useAutoSave';
import { useWebcam } from '@/hooks/useWebcam';
import { useRealtimeFlags } from '@/hooks/useRealtimeFlags';
import { useProctoring } from '@/hooks/useProctoring';
import { Timer } from '@/components/exam/Timer';\nimport { MCQOptions } from '@/components/exam/MCQOptions';\nimport { createSupabaseBrowserClient } from '@/lib/supabase/client';\nimport { toast } from 'sonner';\n```\n\n## Step 2: Initialize Hooks in Component Body\n\n```typescript\nexport default function ExamPage({ params }: { params: Promise<{ id: string }> }) {\n  const router = useRouter();\n  const [resolvedParams, setResolvedParams] = useState<{ id: string } | null>(null);\n  const supabase = createSupabaseBrowserClient();\n  const videoRef = useRef<HTMLVideoElement>(null);\n\n  // Resolve next.js 16 params\n  useEffect(() => {\n    params.then(setResolvedParams);\n  }, [params]);\n\n  if (!resolvedParams) return <div>Loading...</div>;\n\n  const { id: examId } = resolvedParams;\n\n  // State for exam data\n  const [sessionId, setSessionId] = useState<string>('');\n  const [durationMinutes, setDurationMinutes] = useState<number>(120);\n  const [questions, setQuestions] = useState<any[]>([]);\n  const [currentQuestionIndex, setCurrentQuestionIndex] = useState<number>(0);\n  const [answers, setAnswers] = useState<Record<string, string>>({});\n  const [isSubmitting, setIsSubmitting] = useState<boolean>(false);\n\n  // Initialize timer hook\n  const { timeLeft, isExpired, timeString } = useExamTimer(\n    durationMinutes,\n    () => {\n      // Time expired callback\n      toast.error('Time expired! Submitting exam automatically...');\n      handleAutoSubmit();\n    },\n    (minutesRemaining) => {\n      // Warning callback (at 5min and 1min)\n      if (minutesRemaining === 5) {\n        toast.warning('⏰ 5 minutes remaining!');\n      } else if (minutesRemaining === 1) {\n        toast.warning('⏰ Less than 1 minute remaining!');\n      }\n    }\n  );\n\n  // Initialize auto-save hook\n  const { lastSaved, isSaving, saveNow } = useAutoSave(\n    sessionId,\n    examId,\n    answers,\n    30000 // Auto-save every 30 seconds\n  );\n\n  // Initialize webcam hook\n  const { stream, error: webcamError, isPermissionDenied, requestAccess, stopStream } = useWebcam();\n\n  // Initialize realtime flags hook\n  const { flags } = useRealtimeFlags(examId);\n\n  // Initialize proctoring hook\n  const { isInitialized: isProctoringReady, lastFlag, flagCount } = useProctoring(\n    videoRef,\n    sessionId,\n    examId,\n    true // isActive - set to false if exam paused\n  );\n\n  // Step 3: Attach video stream to ref\n  useEffect(() => {\n    if (stream && videoRef.current) {\n      videoRef.current.srcObject = stream;\n    }\n  }, [stream]);\n\n  // Step 4: Load exam session and questions\n  useEffect(() => {\n    const loadExam = async () => {\n      try {\n        // Fetch or create exam session\n        const { data: session } = await supabase.rpc('start_exam_session', {\n          exam_id_param: examId,\n        });\n\n        if (!session) {\n          throw new Error('Failed to start exam session');\n        }\n\n        setSessionId(session.id);\n        setDurationMinutes(session.duration_minutes || 120);\n\n        // Fetch questions\n        const { data: questionsList } = await supabase.rpc('get_exam_questions', {\n          exam_id_param: examId,\n        });\n\n        setQuestions(questionsList || []);\n\n        // Request webcam access\n        await requestAccess();\n      } catch (err) {\n        console.error('Failed to load exam:', err);\n        toast.error('Failed to load exam. Redirecting...');\n        router.push('/student/dashboard');\n      }\n    };\n\n    if (resolvedParams) {\n      loadExam();\n    }\n  }, [resolvedParams, supabase, requestAccess, router]);\n\n  // Step 5: Auto-submit handler\n  const handleAutoSubmit = useCallback(async () => {\n    setIsSubmitting(true);\n    try {\n      // Final save\n      const response = await fetch('/api/submit-exam', {\n        method: 'POST',\n        headers: { 'Content-Type': 'application/json' },\n        body: JSON.stringify({\n          session_id: sessionId,\n          exam_id: examId,\n          answers,\n          auto_submit: true,\n        }),\n      });\n\n      if (!response.ok) throw new Error('Submit failed');\n\n      toast.success('Exam submitted successfully!');\n      stopStream();\n      router.push(`/student/results/${sessionId}`);\n    } catch (err) {\n      console.error('Auto-submit failed:', err);\n      toast.error('Failed to submit exam');\n    } finally {\n      setIsSubmitting(false);\n    }\n  }, [sessionId, examId, answers, stopStream, router]);\n\n  // Step 6: Manual submit handler\n  const handleSubmit = useCallback(async () => {\n    if (confirm('Are you sure you want to submit the exam?')) {\n      await handleAutoSubmit();\n    }\n  }, [handleAutoSubmit]);\n\n  // Step 7: Answer selection handler\n  const handleAnswerChange = (questionId: string, value: string) => {\n    setAnswers((prev) => ({\n      ...prev,\n      [questionId]: value,\n    }));\n  };\n\n  // Step 8: Current question\n  const currentQuestion = questions[currentQuestionIndex];\n\n  if (!sessionId) return <div>Initializing exam...</div>;\n\n  return (\n    <div className=\"min-h-screen bg-gray-50 p-4\">\n      <div className=\"max-w-7xl mx-auto grid grid-cols-3 gap-6\">\n        {/* Left: Webcam */}\n        <div className=\"col-span-1\">\n          <div className=\"bg-white rounded-lg shadow p-4\">\n            <h3 className=\"font-semibold mb-2\">Webcam Feed</h3>\n            {webcamError ? (\n              <div className=\"bg-red-50 text-red-700 p-3 rounded text-sm\">\n                {webcamError}\n                <button\n                  onClick={requestAccess}\n                  className=\"block mt-2 text-red-700 underline\"\n                >\n                  Retry\n                </button>\n              </div>\n            ) : (\n              <video\n                ref={videoRef}\n                autoPlay\n                playsInline\n                muted\n                className=\"w-full rounded bg-black\"\n              />\n            )}\n            <div className=\"mt-3 text-xs space-y-1\">\n              <p>🎥 Proctoring: {isProctoringReady ? '✅ Active' : '⏳ Loading...'}</p>\n              <p>🚩 Flags: {flagCount}</p>\n              {lastFlag && <p>Last: {lastFlag}</p>}\n            </div>\n          </div>\n        </div>\n\n        {/* Center: Questions */}\n        <div className=\"col-span-2 bg-white rounded-lg shadow p-6\">\n          {currentQuestion && (\n            <div>\n              <div className=\"mb-6\">\n                <h2 className=\"text-xl font-semibold\">\n                  {currentQuestionIndex + 1}. {currentQuestion.text}\n                </h2>\n                <p className=\"text-sm text-gray-600 mt-1\">\n                  Type: {currentQuestion.type}\n                </p>\n              </div>\n\n              {currentQuestion.type === 'mcq' ? (\n                <MCQOptions\n                  options={JSON.parse(currentQuestion.options || '[]')}\n                  selectedValue={answers[currentQuestion.id] || null}\n                  onChange={(value) => handleAnswerChange(currentQuestion.id, value)}\n                />\n              ) : (\n                <textarea\n                  value={answers[currentQuestion.id] || ''}\n                  onChange={(e) => handleAnswerChange(currentQuestion.id, e.target.value)}\n                  className=\"w-full h-40 p-3 border rounded focus:ring-2 focus:ring-blue-400 outline-none\"\n                  placeholder=\"Enter your answer here...\"\n                />\n              )}\n\n              {/* Navigation */}\n              <div className=\"flex gap-3 mt-8\">\n                <button\n                  onClick={() => setCurrentQuestionIndex(Math.max(0, currentQuestionIndex - 1))}\n                  disabled={currentQuestionIndex === 0}\n                  className=\"px-4 py-2 bg-gray-100 rounded disabled:opacity-50\"\n                >\n                  ← Previous\n                </button>\n                <div className=\"flex-1\" />\n                <div className=\"text-sm text-gray-600 py-2\">\n                  Question {currentQuestionIndex + 1} of {questions.length}\n                </div>\n                <button\n                  onClick={() => setCurrentQuestionIndex(Math.min(questions.length - 1, currentQuestionIndex + 1))}\n                  disabled={currentQuestionIndex === questions.length - 1}\n                  className=\"px-4 py-2 bg-blue-100 text-blue-700 rounded disabled:opacity-50\"\n                >\n                  Next →\n                </button>\n              </div>\n            </div>\n          )}\n        </div>\n      </div>\n\n      {/* Top Bar: Timer + Controls */}\n      <div className=\"fixed top-0 left-0 right-0 bg-white border-b p-4 shadow-sm\">\n        <div className=\"max-w-7xl mx-auto flex justify-between items-center\">\n          <div className=\"flex items-center gap-6\">\n            <Timer\n              timeString={timeString}\n              isExpired={isExpired}\n              durationMinutes={durationMinutes}\n              currentSeconds={timeLeft}\n            />\n          </div>\n\n          <div className=\"flex items-center gap-4 text-sm\">\n            <span className=\"text-gray-600\">\n              Last saved: {lastSaved ? lastSaved.toLocaleTimeString() : 'Never'}\n            </span>\n            <button\n              onClick={saveNow}\n              disabled={isSaving}\n              className=\"px-3 py-1 bg-blue-50 text-blue-700 rounded text-sm\"\n            >\n              {isSaving ? 'Saving...' : 'Save Now'}\n            </button>\n            <button\n              onClick={handleSubmit}\n              disabled={isSubmitting || isExpired}\n              className=\"px-4 py-2 bg-green-600 text-white rounded hover:bg-green-700 disabled:opacity-50\"\n            >\n              {isSubmitting ? 'Submitting...' : 'Submit Exam'}\n            </button>\n          </div>\n        </div>\n      </div>\n    </div>\n  );\n}\n```\n\n## Key Integration Points\n\n### 1. Timer Hook\n- Returns: `{timeLeft, isExpired, timeString}`\n- Use `timeLeft` for progress calculations\n- Use `timeString` for display (already formatted as MM:SS)\n- Use `isExpired` to disable submit button\n\n### 2. Auto-Save Hook\n- Automatically saves every 30 seconds\n- Call `saveNow()` for manual save\n- Use `lastSaved` to show user when last save occurred\n- Use `isSaving` to disable UI during save\n\n### 3. Webcam Hook\n- Must call `requestAccess()` to start video stream\n- Attach `stream` to video element via `srcObject`\n- Handle `webcamError` and `isPermissionDenied` states\n- Call `stopStream()` on cleanup or exam submit\n\n### 4. Proctoring Hook\n- Pass `videoRef` from the video element\n- Proctoring is active by default; set 4th param to false to pause\n- Returns: `{isInitialized, lastFlag, flagCount}`\n- Flags are posted to `/api/flag-event` automatically\n\n### 5. Realtime Flags Hook\n- Subscribes to live flag events via Supabase Realtime\n- Useful for displaying live flag feed or flag count\n- Returns: `{flags: Flag[]}`\n\n### 6. MCQ Options Component\n- Pass array of `{value, label}` objects\n- Set `selectedValue` to currently selected option value\n- Handle `onChange` callback to update answer state\n- Renders with A/B/C/D labels automatically\n\n### 7. Timer Component\n- Props: `timeString`, `isExpired`, `durationMinutes`, `currentSeconds`\n- Shows MM:SS display with dynamic colors\n- Color states: gray (normal) → yellow (warning) → red (critical) → red-600 (expired)\n- Progress bar shows time remaining\n\n## Required RPC Functions\n\nEnsure these functions exist in Supabase:\n\n```sql\n-- Start exam session\nCREATE OR REPLACE FUNCTION start_exam_session(exam_id_param uuid)\nRETURNS TABLE (id uuid, user_id uuid, exam_id uuid, started_at timestamp, duration_minutes int)\nLANGUAGE sql\nSECURITY DEFINER\nAS $$\n  -- Implementation: Create session, return created record\n$$;\n\n-- Get exam questions\nCREATE OR REPLACE FUNCTION get_exam_questions(exam_id_param uuid)\nRETURNS TABLE (id uuid, text text, type text, options text, correct_answer text, max_marks int)\nLANGUAGE sql\nSECURITY DEFINER\nAS $$\n  SELECT id, text, type, options, correct_answer, max_marks\n  FROM questions\n  WHERE exam_id = exam_id_param\n  ORDER BY position ASC;\n$$;\n```\n\n## Environment Variables Required\n\n```\n# .env.local\nNEXT_PUBLIC_SUPABASE_URL=your_url\nNEXT_PUBLIC_SUPABASE_ANON_KEY=your_key\nNEXT_PUBLIC_GEMINI_API_KEY=your_gemini_key (for AI detection)\nDAILY_API_KEY=optional (for video rooms)\n```\n\n## Common Issues\n\n### Webcam Not Showing\n- Check browser console for permission errors\n- Ensure camera permissions granted in browser settings\n- Video element must have `ref` attached before stream loads\n\n### Timer Not Counting Down\n- Verify `durationMinutes` is correctly passed\n- Check browser console for useEffect errors\n- Ensure `timeLeft` updates are triggering re-renders\n\n### Proctoring Not Detecting\n- Verify Face-API models are in `/public/models/`\n- Check browser console for model loading errors\n- Ensure video stream is playing (check `stream` state)\n\n### Auto-Save Not Working\n- Verify `/api/submit-exam` endpoint exists\n- Check network tab for POST requests\n- Ensure `sessionId` and `examId` are set before auto-save starts\n\n## Next Steps\n\n1.
